ambition_become_sotapanna = {
	can_be_picked = {
		NOT = {
			has_variable = ongoing_scheme_target
		}
		OR = {
			has_any_position = yes
			is_ruler = yes			
		}
	}
	
	finished_when = {
		custom_tooltip = {
			text = "scheme_ambition_fulfillment_always_yes"
		}
	}
	
	duration = 365
	
	abort = {
		NOT = { has_religion = buddhism }
		OR = {
			is_alive = no
			has_character_modifier = indisposed_charmod
			prisoner = yes
		}		
	}
	
	chance = {
		base = 10
	}
	on_start = ambition_become_sotapanna_start
	on_finish = ambition_inward_journey_finish
	on_abort = scheme_abort

	modifier = {
		monthly_character_wealth = -6
	}
}
