inward_journey = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_smear_reputation"
	
	potential_trigger = {
		hidden:scope:target = {
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			is_alive = yes
			is_ruler = yes
			has_religion = buddhism
		}
	}
	
	
	allowed_trigger = {
		hidden:scope:actor = {
			can_pay_price = inward_journey_price
		}
		scope:target = {
			trigger_if = { 
				limit = { is_adult = no }
				is_adult = yes
			}
			popularity > 5
			NOT = {
				has_character_modifier = commander_king
			}
			trigger_if = {
				limit = {
					has_variable = meditation_cooldown
				}
				custom_tooltip = {
					text = "meditation_char_interaction_tooltip"
					NOT = {
						has_variable = meditation_cooldown
					}
				}
			}			
		}
	}
	

	
	effect = {
		scope:actor = {
			if = {
				limit = { is_ai = no }
				pay_price = inward_journey_price
			}
			trigger_event = {
				id = hindu_flavor.2
			}
		}
		scope:target = {
			add_popularity = -5
			set_variable = {
				name = meditation_cooldown
				days = 1825	#5 Years
			}
		}
	}
	
	ai_will_do = {
		modifier = {
			factor = 5
		}
	}
	
}
