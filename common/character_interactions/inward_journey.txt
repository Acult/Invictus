inward_journey = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_smear_reputation"
	
	potential_trigger = {
		hidden:scope:target = {
			NOT = { has_trait = samyaksambodhi }
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			is_alive = yes
			OR = {
			has_any_position = yes
			is_ruler = yes
			}
			has_religion = buddhism
		}
	}
	
	
	allowed_trigger = {
		hidden:scope:actor = {
			can_pay_price = inward_journey_price
		}
		scope:target = {
			trigger_if = { 
				limit = { is_adult = no }
				is_adult = yes
			}
			popularity > 5
			NOT = {
				has_character_modifier = commander_king
			}
			trigger_if = {
				limit = {
					has_variable = meditation_cooldown
				}
				custom_tooltip = {
					text = "meditation_char_interaction_tooltip"
					NOT = {
						has_variable = meditation_cooldown
					}
				}
			}			
		}
	}
	

	
	effect = {
		scope:actor = {
			if = {
				limit = { is_ai = no }
				pay_price = inward_journey_price
			}
		}
		scope:target = {
			save_scope_as = ongoing_scheme_target
			if = {
				limit = {
					exists = scope:ongoing_scheme_target
				}
				set_variable = {
					name = ongoing_scheme_target
					value = scope:ongoing_scheme_target
					days = 365	#If this value is changed all the ambitions also have to be changed
				}
			}		
			add_popularity = -5
#			set_variable = {
#				name = meditation_cooldown
#				days = 1825	#5 Years
#			}
			if = {
				limit = {
					scope:target = {
						NOR = {
							has_trait = sotapanna
							has_trait = sakadagami
							has_trait = anagami
							has_trait = arhat
							has_trait = samyaksambodhi						
						}
					}
				}
			set_ambition = ambition_become_sotapanna
			}
			else_if = {
				limit = {
					scope:target = {
						has_trait = sotapanna
						NOR = {
							has_trait = sakadagami
							has_trait = anagami
							has_trait = arhat
							has_trait = samyaksambodhi	
						}
					}
				}
			set_ambition = ambition_become_sakadagami				
			}
			else_if = {
				limit = {
					scope:target = {
						has_trait = sakadagami
						NOR = {
							has_trait = sotapanna
							has_trait = anagami
							has_trait = arhat
							has_trait = samyaksambodhi	
						}
					}
				}
			set_ambition = ambition_become_anagami				
			}
			else_if = {
				limit = {
					scope:target = {
						has_trait = anagami
						NOR = {
							has_trait = sotapanna
							has_trait = sakadagami
							has_trait = arhat
							has_trait = samyaksambodhi	
						}
					}
				}
			set_ambition = ambition_become_arhat		
			}
			else = {
				limit = {
					scope:target = {
						has_trait = arhat
						NOR = {
							has_trait = sotapanna
							has_trait = sakadagami
							has_trait = anagami
							has_trait = samyaksambodhi	
						}
					}
				}
			set_ambition = ambition_become_samyaksambodhi
			}									
		}
	}
	
	ai_will_do = {
		modifier = {
			factor = 5
		}
	}
	
}
