#RHODES MECHANIC SUPPORT EFFECTS, ONLY USE THESE EFFECTS TO CHANGE SUPPORT
faction_support_cleanup_effect = {
	#Run this effect after adding support to make sure the value doesn't go over 30 or under 0
	if = {
		limit = { var:merchant_support > 30 }
		remove_variable = merchant_support
		set_variable = {
			name = merchant_support
			value = 30
		}
	}
	else = {
		limit = { var:merchant_support < 0 }
		remove_variable = merchant_support
		set_variable = {
			name = merchant_support
			value = 0
		}
	}
	if = {
		limit = { var:military_support > 30 }
		remove_variable = military_support
		set_variable = {
			name = military_support
			value = 30
		}
	}
	else = {
		limit = { var:military_support < 0 }
		remove_variable = military_support
		set_variable = {
			name = military_support
			value = 0
		}
	}
}
add_10_merchant_support_effect = {
	custom_tooltip = add_10_merchant_support_tt
	if = {
		limit = { has_variable = merchant_support }
		change_variable = {
			name = merchant_support
			add = 10
		}
	}
	else = {
		limit = { 
			NOT = { has_variable = merchant_support }
		}
		set_variable = {
			name = merchant_support
			value = 10
		}
	}
}
