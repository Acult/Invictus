scheme_ruler_legitimacy_finish = {
	effect = {
		employer = {
			add_legitimacy = 5
		}

			limit = {
				has_culture_group = hellenic
				is_male = yes
				NOT = {
					has_bloodline_trigger = yes
				}
				number_of_traits <= 7
			}
			random = {
				chance = 2
				custom_tooltip = gain_random_diadochi_bloodline_trait
				hidden_effect = {
					trigger_event = {
						id = scheme.11
					}
				}
			}
		}
		if = {
			limit = {
				OR = {
					has_culture_group = aryan
					has_culture_group = indian
					has_culture_group = pracyan
				}
				is_male = yes
				NOT = {
					has_bloodline_trigger = yes
				}
				number_of_traits <= 7
			}
			random = {
				chance = 2
				custom_tooltip = gain_random_diadochi_bloodline_trait
				hidden_effect = {
					trigger_event = {
						id = scheme.11
					}
				}
			}
		}
		remove_scheme_variables_effect = yes
	}
}

scheme_convert_pops_in_holding_start = {
	effect = {
		save_scope_as = ongoing_scheme_target
		set_variable = {
			name = ongoing_scheme_target
			value = scope:ongoing_scheme_target
			days = 400
		}
		add_character_modifier = {
			name = "encroaching_on_public_land_modifier"
			duration = 365
		}
		set_variable = {
			name = converting_locals
			days = 1825
		}
		random_holdings = {
			limit = {
				any_pops_in_province = {
					count >= 15
					NOT = { pop_religion = root.religion }
				}
			}
			save_scope_as = converting_holding
		}
	}	
}
scheme_convert_pops_in_holding_finish = {
	effect = {
		scope:converting_holding = {
			random_list = {
				33 = {
                    modifier = {
						scope:ongoing_scheme_target.zeal <= 5
						factor = 15
                    }					
				}
				33 = {
                    modifier = {
                    	scope:ongoing_scheme_target.zeal < 10
                    	scope:ongoing_scheme_target.zeal > 5
                        factor = 10
                    }
					random_pops_in_province = {
						limit = {
							NOT = { pop_religion = root.religion }
						}
						set_pop_religion = root.religion
					}
				}
				33 = {
                    modifier = {
                    	scope:ongoing_scheme_target.zeal >= 10
                        factor = 10
                    }
					ordered_pops_in_province = {
						limit = {
							NOT = { pop_religion = root.religion }
						}
						order_by = pop_hapiness
						max = 2
						set_pop_religion = root.religion
					}
				}
			}
		}
		hidden_effect = {
			remove_scheme_variables_effect = yes
		}
	}
}

scheme_exploit_populous_finish = {
	effect = {
		hidden_effect = {
			scope:exploiting_holding = { 
				set_variable = {
					name  = holding_has_been_exploited
					days = 9125
				}
			}
			random_list = {
				20 = {
					scope:exploiting_holding = {
						ordered_pops_in_province = {
							order_by = pop_hapiness
							max = {
								value = total_population
								multiply = 0.1
								round = floor
							}
							check_range_bounds = no
							kill_pop = yes
						}
					}
				}
				20 = {
					scope:exploiting_holding = {
						while = {
							count = 2
							random_pops_in_province = { kill_pop = yes }
						}
					}
				}
				20 = {
					scope:exploiting_holding = {
						random_pops_in_province = { kill_pop = yes }
					}
				}
				20 = {
					scope:exploiting_holding = {
						while = {
							count = 3
							random_pops_in_province = { kill_pop = yes }
						}
					}
				}
				20 = {
					scope:exploiting_holding = {
						random_pops_in_province = { kill_pop = yes }
						add_province_modifier = {
							name = devestation
							duration = 720
						}
					}
				}
			}
		}
	}
}

scheme_training_skill_finish = {
	effect = {
		hidden_effect = {
			if = {
				limit = { has_variable = finesse_training }
				random_list = {
					2 = { add_finesse = 1 }
					2 = { add_finesse = 2 }
					1 = { 
						limit = { finesse < 10}
						add_finesse = 3 
					}
					1 = { add_finesse = -1 }
					2 = {
						limit = {
							NOT = { has_trait = crafty }
							number_of_traits <= 7							
						}
						force_add_trait = crafty
					}
					2 = {
						limit = {
							NOT = { has_trait = intelligent }
							number_of_traits <= 7							
						}
						force_add_trait = intelligent						
					}
				}
			}
			else_if = {
				limit = { has_variable = zeal_training }
				random_list = {
					2 = { add_zeal = 1 }
					2 = { add_zeal = 2 }
					1 = { 
						limit = { zeal < 10}
						add_zeal = 3 
					}
					1 = { add_zeal = -1 }
					2 = {
						limit = {
							NOT = { has_trait  = obsessive}
							number_of_traits <= 7
						}
						force_add_trait = obsessive
					}
					2 = {
						limit = {
							NOT = { has_trait  = zealous }
							number_of_traits <= 7
						}
						force_add_trait = zealous
					}
				}
			}
			else_if = {
				limit = { has_variable = charisma_training }
				random_list = {
					2 = { add_charisma = 1 }
					2 = { add_charisma = 2 }
					1 = { 
						limit = { charisma < 10}
						add_charisma = 3 
					}
					1 = { add_charisma = -1 }
					2 = {
						limit = {
							NOT = { has_trait  = humble }
							number_of_traits <= 7
						}
						force_add_trait = humble
					}
					2 = {
						limit = {
							NOT = { has_trait  = silver_tongued }
							number_of_traits <= 7
						}
						force_add_trait = silver_tongued			
					}
				}
			}
			else_if = {
				limit = { has_variable = martial_training }
				random_list = {
					2 = { add_martial = 1 }
					2 = { add_martial = 2 }
					1 = { 
						limit = { martial < 10}
						add_martial = 3 
					}
					1 = { add_martial = -1 }
					2 = {
						limit = {
							NOT = { has_trait = disciplined }
							number_of_traits <= 7							
						}
						force_add_trait = disciplined
					}
					2 = {
						limit = {
							NOT = { has_trait = brave }
							number_of_traits <= 7							
						}
						force_add_trait = brave						
					}
				}
			}
		}
	}
