country_decisions = {
	
	# Form Achaean League
	form_achaean_league = {
		
		potential = {
			num_of_cities >= 1
			OR = {
				primary_culture = achaean
				primary_culture = arcadian
			}
			is_republic = yes
			NOT = {
				tag = ACL 			### Tag name of Achaean League
			}
			NOT = {
				tag = HLA
			}
			NOR = {
				is_endgame_tag_trigger = yes
				is_tier_2_formable_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {
				province_id = 8003  #Sckyon
			}
		}
		
		allow = {
			custom_tooltip = {
				text = "formable_not_achean_league_exists"		
				NOT = {
					any_country = {
						tag = ACL
					}
				}
			}
			can_form_nation_trigger = yes
			NOT = {
				OR = {
					c:PRY = {
						owns = 8003
					}
					c:EGY = {
						owns = 8003
					}
					c:MAC = {
						owns = 8003
					}
					c:TRE = {
						owns = 8003
					}
					c:SEL = {
						owns = 8003
					}
				}
			}
		}
		
		effect = {
			custom_tooltip = "tier_2_formable_tooltip"
			change_country_name = "ACL_FORMABLE_NAME"  
			hidden_effect = {
				change_country_adjective = "ACL_FORMABLE_ADJ"	
				change_country_color =  "achaean_league_color"			
				change_country_tag = ACL
				change_country_flag = ACHAEAN_LEAGUE_FLAG				
			}
			capital_scope = {
				capital_formable_medium_effect = yes
				formable_capital_modifier_normal_effect = yes
			}
			add_2_free_province_investments = yes
			trigger_event = { id = dhe_achaean_league.1	}
			custom_tooltip = "form_achaean_league_claims_tooltip"	
			hidden_effect = {
				area:achaea_area = {
					add_provincial_claim_effect = yes
				}
				area:arcadia_area = {
					add_provincial_claim_effect = yes
				}
			}
		}
		
		ai_will_do = {
			base = 100
		}
	}
}
