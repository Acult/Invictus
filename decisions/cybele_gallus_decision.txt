country_decisions = {
	make_gallus = {
		potential = {
			num_of_cities >= 1
			religion = anatolian_religion
		}
		
		allow = {
			any_character = {
				trigger_if = {
				   limit = {
					  root = { is_monarchy = yes }
				   }
				   has_office = office_high_priest_monarchy
				   NOT = { has_trait = gallus }
				}
				trigger_if = {
				   limit = {
					  root = { is_republic = yes }
				   }
				   has_office = office_pontifex
				   NOT = { has_trait = gallus }
				}
				trigger_if = {
				   limit = {
					  root = { is_tribal = yes }
				   }
				   has_office = office_high_priest
				   NOT = { has_trait = gallus }
				}
			}
		}
		effect = {
			random_character = {
				limit = {
				   root = { is_monarchy = yes }
				   has_office = office_high_priest_monarchy
				}
				force_add_trait = gallus
			}
			random_character = {
				limit = {
				   root = { is_republic = yes }
				   has_office = office_pontifex
				}
				force_add_trait = gallus
			}
			random_character = {
				limit = {
				   root = { is_tribal = yes }
				   has_office = office_high_priest
				}
				force_add_trait = gallus
			}
		}
	}
	become_gallus = {
		potential = {
			num_of_cities >= 1
			religion = anatolian_religion
			current_ruler = {
				zeal >= 8
			}
		}
		allow = {
			current_ruler = {
				zeal >= 8
				NOT = {
					has_trait = gallus
				}
			}
		}
		effect = {
			force_add_trait = gallus
		}
	}
}
