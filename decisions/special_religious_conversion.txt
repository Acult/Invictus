		}

			political_influence >= 100

				limit = { is_republic = yes }
				has_senate_approval >= 0.15
			}

					count = all

						is_clan_chief = no

					}

			}

				has_country_modifier = recent_convert_modifier
			}
			custom_tooltip = {
				text = "Have 3 deities from [GetReligion('roman_pantheon').GetName]."

					count >= 3
					deity_religion = roman_pantheon
				}
			}
		}

		effect = {
			add_political_influence = -100
			trigger_if = { #monarchy
				limit = { is_monarchy = yes }
				add_legitimacy = -35
			}
			trigger_else_if = { #Republic
				limit = { is_republic = yes }
				add_senate_approval = -0.15

			}
