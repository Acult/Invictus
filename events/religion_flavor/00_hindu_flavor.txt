hindu_flavor.2 = {
	type = country_event
	title = hindu_flavor_2_title
	desc = hindu_flavor_2_desc
	picture = pataliputra
	
	left_portrait = current_ruler
	right_portrait = current_ruler	

	immediate = {
		current_ruler = {
			force_add_trait = meditating
			save_scope_as = meditating_ruler
		}
	}

	option = {
		name = "hindu_flavor_2.a"
		##A short Meditation (2 weeks)
		set_variable = {
			name = two_week_meditation
			days = 365
		}
		trigger_event = {
			id = hindu_flavor.3
			days = 14
		}			
	}

	option = {
		name = "hindu_flavor_2.b"
		##A lengthy Meditation (3 months)
		set_variable = {
			name = three_month_meditation
			days = 365
		}
		trigger_event = {
			id = hindu_flavor.3
			days = 90
		}				
	}	

	option = {
		name = "hindu_flavor_2.c"
		##A Seculuded Meditation (6 months)
		set_variable = {
			name = six_month_meditation
			days = 365
		}
		trigger_event = {
			id = hindu_flavor.3
			days = 180
		}				 
	}

}
