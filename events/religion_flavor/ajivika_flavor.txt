namespace = ajivika_flavor

ajivika_flavor.1 = {
	type = country_event
	title = ajivika_flavor.1.t
	desc = ajivika_flavor.1.desc
	picture = revolt

	left_portrait = current_heir
	fire_only_once = yes

	trigger = {
		current_ruler = {
			RELIGION = buddhism
		}
		tag = MRY
	}

	immediate = {
		add_stability = -3
		capital_scope = { #Change to Pundravardhana
			random_pops_in_province = {
				limit = {
					pop_religion = buddhism
				}
				while = {
					count = 1
					kill_pop = yes
				}
			}
		}
	}

	option = {
		name = ajivika_flavor.1.a
		hidden_effect = {
			trigger_event = {
				id = ajivika_flavor.2
				days = { 100 200 }
			}
		}
		capital_scope = { #Change to Pundravardhana
			random_pops_in_province = {
				limit = {
					pop_religion = ajivika
				}
				while = {
					count = 4
					kill_pop = yes
				}
			}
		}
		add_stability = 6
		add_political_influence = 30
	}
	option = {
		name = ajivika_flavor.1.b
		capital_scope = {
			while = {
				count = 2
				random_pops_in_province = {
					limit = {
						pop_religion = buddhism
					}
					set_pop_religion = ajivika
				}
			}
		}
		add_stability = -3
		add_political_influence = -30
		add_legitimacy = subtract_legitimacy_small
	}
}
