namespace = zorgo_event

zorgo_event.1 = {
	type = country_event
	title = "zorgo_event.1.t"
	desc = "zorgo_event.1.desc"
	left_portrait = current_ruler
	right_portrait = scope:random_dude
	right_portrait = scope:random_dude_2

	picture = assassination
	
	trigger = {
		current_date > 500.1.1
		is_subject = no
	}
	
	immediate = {
		random_character = {
			limit = {
				is_head_of_family = no
				is_ruler = no
				is_adult = yes
				is_male = yes
				from_ruler_family = yes
			}
			save_scope_as = random_dude
		}
		random_character = {
			limit = {
				is_head_of_family = no
				is_ruler = no
				is_adult = yes
				is_male = yes
				from_ruler_family = no
			}
			save_scope_as = random_dude_2
			force_add_trait = one_eyed
		}
	}

	fire_only_once = yes

	option = {
		name = "zorgo_event.1.a"
		add_treasury = -20
	}
	
	option = {
		name = "zorgo_event.1.b"
		add_tyranny = 3
	}
}

			}
			save_scope_as = dumbass
		}
	}
	
	fire_only_once = yes

	option = {
		name = "zorgo_event.3.a"
		add_stability = 5
		add_treasury = -20
	}
	
	option = {
		name = "zorgo_event.3.b"
		add_tyranny = 1
	}

	option = {
		highlight = yes
		current_ruler = {
			add_trait = cruel
		}
		name = "zorgo_event.3.c"
		add_tyranny = 5
		scope:dumbass = {
			death = {
				killer = current_ruler.root
				death_reason = death_murder
			}
		}
	}
}
