me_lanka.2 = { # Food Imports - Make a province produce a trade good of your choice
    type = country_event
    title = me_lanka_2_title
    desc = me_lanka_2_desc
    picture = interesting_histories_southern_india

    option = {	#Grain
        name = me_lanka_2.a
        set_variable = {
        	name = grain_chosen
        	days = 180
        }
    }

    option = {	#Vegetables
        name = me_lanka_2.b
        set_variable = {
        	name = vegetables_chosen
        	days = 180
        }
    }
    option = {	#Livestock
        name = me_lanka_2.c
        set_variable = {
        	name = livestock_chosen
        	days = 180
        }
    }

    option = {	#Sugar
        name = me_lanka_2.d
        trigger = {
        	capital_scope = {
        		is_importing_trade_good = sugar
        	}
        }
        set_variable = {
        	name = sugar_chosen
        	days = 180
        }
    }

    after = {
    	trigger_event = {
    		id = me_lanka.3
    	}
    }        
}

me_lanka.5 = {  #The Question of Ruhunu
    type = country_event
    title = me_lanka_5_title
    desc = me_lanka_5_desc
    picture = interesting_histories_southern_india

    option = {  #Integrate them immediately
        name = me_lanka_5.a
        c:RHN = {
            every_owned_province = {
                set_owned_by = root
            }
        }
    }

    option = {  #A feudatory would be nice
        name = me_lanka_5.b
        c:RHN = {
            make_subject = {
                target = ROOT
                type = feudatory
            }
        }
    }    
}

me_lanka.6 = {  #Dharmashoka
    type = country_event
    title = me_lanka_6_title
    desc = me_lanka_6_desc
    picture = interesting_histories_southern_india

    trigger = {
        NOT = {
            current_ruler = {
                is_friend = c:MRY.current_ruler
            }
        }
    }

    option = {  #Befreind Ashoka with gifts
        name = me_lanka_6.a
        add_treasury = -100
        current_ruler = {
            add_friend = char:66
            set_character_religion = buddhism
        }
    }

    option = {  #Befreind Ashoka with persuasion
        name = me_lanka_6.b
        add_political_influence = -25
        current_ruler = {
            add_friend = char:66
            set_character_religion = buddhism
        }
    }    
}
