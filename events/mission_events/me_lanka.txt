me_lanka.2 = { # Food Imports - Make a province produce a trade good of your choice
    type = country_event
    title = me_lanka_2_title
    desc = me_lanka_2_desc
    picture = interesting_histories_southern_india

    option = {	#Grain
        name = me_lanka_2.a
        set_variable = {
        	name = grain_chosen
        	days = 180
        }
    }

    option = {	#Vegetables
        name = me_lanka_2.b
        set_variable = {
        	name = vegetables_chosen
        	days = 180
        }
    }
    option = {	#Livestock
        name = me_lanka_2.c
        set_variable = {
        	name = livestock_chosen
        	days = 180
        }
    }

    option = {	#Sugar
        name = me_lanka_2.d
        trigger = {
        	capital_scope = {
        		is_importing_trade_good = sugar
        	}
        }
        set_variable = {
        	name = sugar_chosen
        	days = 180
        }
    }

    after = {
    	trigger_event = {
    		id = me_lanka.3
    	}
    }        
}
