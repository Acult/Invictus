me_colchis.14 = { # Guarding Roses
    type = country_event
    title = me_colchis_14_title
    desc = me_colchis_14_desc 
	picture = city_construction
	left_portrait = current_ruler
    right_portrait = scope:colchis_rose_general
	trigger = {
		always = yes
	}

    immediate = {
        create_character = { 
            female = no
            age = 34
            culture = root.culture
            religion = root.religion
            no_stats = yes
            hidden_effect = {
                amazing_martial_effect = yes
                great_charisma_effect = yes
                bad_finesse_effect = yes
                bad_zeal_effect = yes
                add_trait = ambitious
                add_trait = sceptical
                add_trait = intelligent
                save_scope_as = colchis_rose_general
            }
        }
    }

	option = { # Make the Fortress Profitable
		name = me_colchis_14.a
        add_treasury = -150
        current_ruler = {
            add_gold = 150
            scope:colchis_rose_general = {
                add_gold = 150
            }
        }
        p:1706 = {
            while = {
                count = 4
                define_pop = {
                    type = slaves
                    religion = root.religion
                    culture = root.culture
                }
            }
            add_permanent_province_modifier = {
                name = rosy_mines
            }
        }
        hidden_effect = {
            random = {
                chance = 35
                trigger_event = {
                    id = me_colchis.15
                    months = { 12 36 }
                }
            }
        }
	}

    option = { # Make more food
		name = me_colchis_14.b
        add_manpower = -7
        p:1706 = {
            set_trade_goods = grain
            add_permanent_province_modifier = {
                name = fields_of_roses
            }
            while = {
                count = 2
                define_pop = {
                    type = citizen
                    religion = root.religion
                    culture = root.culture
                }
            }
        }
	}

    option = { # Take the Mountain Passes
		name = me_colchis_14.c
        add_manpower = -7
        current_ruler = {
            add_friend = scope:colchis_rose_general
        }
        scope:colchis_rose_general = {
            add_loyalty = 5               # This causes error
            add_martial = 1
        }
        p:1700 = {
            set_conquered_by = root
            add_claim = c:IBR
            add_permanent_province_modifier = {
                name = rose_thorn
            }
        }
        p:1703 = {
            set_conquered_by = root
            add_claim = c:IBR
            add_permanent_province_modifier = {
                name = rose_thorn
            }
        }
        p:1704 = {
            set_conquered_by = root
            add_claim = c:IBR
            add_permanent_province_modifier = {
                name = rose_thorn
            }
        }
        c:IBR = {
            add_opinion = {
                target = root
                modifier = siezed_western_passes
            }
        }
	}
}
