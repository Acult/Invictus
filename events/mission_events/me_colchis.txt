me_colchis.9 = { # A Piece of the Past
    type = country_event
    title = me_colchis_9_title
    desc = me_colchis_9_desc 
    picture = throneroom
    left_portrait = current_ruler
    trigger = {
        always = yes
    }

    immediate = {
        random_country = {
            limit = {
                NOT = {war_with = ROOT}
                primary_culture = bosporan
                is_ai = no
            }
            save_scope_as = iason_stone_buyer
        }
    }

    option = { # Treasure the Stone
        name = me_colchis_9.a
        add_political_influence = 50
        add_stability = -5
        current_ruler = {
            add_popularity = -10
        }
        add_tyranny = 5
        random_country_culture = {
            limit = {
                is_culture = colchian
            }
            add_country_culture_modifier = {
                name = anger_against_argonaut_stone
                duration = -1
            }
        }
        create_country_treasure = {
			key = stone_of_iason
			icon = "treasure_it_sacred_stone_of_emesa"
			modifier = {
				local_migration_attraction = 3
                local_migration_speed = 0.1
                local_population_growth = 0.1
			}
		}
        random_province = {
            limit = {
                owner = root
                has_building = port_building
            }
            while = {
                count = 5
                define_pop = {
                    type = citizen
                    religion = religion:roman_pantheon
                    culture = culture:bosporan
                }
            }
        }
        set_variable = stone_iason_worship
    }

    option = { # Ritually Destroy the Stone
        name = me_colchis_9.b
        current_ruler = {                # Should activate a followup decision to deify Aeetes !!!!!!!!
            add_popularity = 10
        }
        add_tyranny = -10
        add_legitimacy = 10
        every_country = {
            limit = {
                primary_culture = bosporan
            }
            add_opinion = {
                modifier = outrage_over_stone_destruction
                target = root
            }
        }
        random_country_culture = {
            limit = {
                is_culture = colchian
            }
            add_country_culture_modifier = {
                name = praise_stone_destruction
                duration = -1
            }
        }
        add_country_modifier = {
            name = destroy_argonaut_stone
        }
    }

    option = { # Sell the Stone
        name = me_colchis_9.c
        add_treasury = 300
        scope:iason_stone_buyer = { #showing up weird in-game / not working !!!!!!!
            current_ruler = {
                add_gold = -300
            }
            create_country_treasure = {
                key = stone_of_iason
                icon = "treasure_it_sacred_stone_of_emesa"
                modifier = {
                    local_migration_attraction = 3
                    local_migration_speed = 0.1
                    local_population_growth = 0.1
                }
            }
        }
    }

    option = { # Dismiss this Hoax
        name = me_colchis_9.d
        current_ruler = {
            add_popularity = -3
        }
    }
}
