me_bithynia.1 = { 
    type = country_event

    title = me_bithynia_1_title
    desc = me_bithynia_1_desc
    picture = great_library

    immediate = {
        ordered_owned_province = {
            limit = {
                has_city_status = yes
                is_capital = no
            }
            order_by = total_population
            max = 1
            save_scope_as = new_capital
        }
    }

    option = { 
        name = me_bithynia_1.a
        custom_tooltip = me_bithynia_1_tooltip
        if = {
            limit = {
                capital_scope = { has_city_status = yes } 
            }
            add_treasury = 100
            add_1_free_province_investments = yes
        }
        else = {
            limit = {
                capital_scope = { has_city_status = no }
            }
            set_capital = scope:new_capital
            add_1_free_province_investments = yes
        }        
    }
}

me_bithynia.59 = { 
    type = country_event

    title = me_bithynia_59_title
    desc = me_bithynia_59_desc
    picture = great_library

    immediate = {
        custom_tooltip = me_bithynia_59_tooltip        
    }
    option = { 
        name = me_bithynia_59.a
        hidden_effect = {
            p:1453 = {
                create_country = {
                    name = {
                        name = byzantium_name
                        adjective = byzantium_adj
                    }
                    #change_country_flag = KUSH_CLIENT_FLAG
                    #set_country_heritage = indo_greek_heritage
                    change_country_color = "eastern_roman_color"
                    change_government = aristocratic_monarchy
                    set_country_religion = roman_pantheon
                    set_primary_culture = bithynian
                    save_scope_as = byz_country_scope
                }
            }
            make_subject = {
                target = scope:byz_country_scope
                type = feudatory
            }
        }
    }
    option = { 
        name = me_bithynia_59.b
        #Add a few more provinces to the list
        custom_tooltip = me_bithynia_59_tooltip_B
        if = {
            limit = {
                OR = {
                    p:1453 = { is_capital = yes }
                    p:345 = { is_capital = yes }
                    p:341 = { is_capital = yes }
                }
            }
            ordered_owned_province = {
                limit = {
                    NOR = {
                        THIS = p:1453
                        THIS = p:345
                        THIS = p:341
                    }
                }
                order_by = total_population
                max = 1
                save_scope_as = new_capital
            }
            set_capital = scope:new_capital
        }
        p:1453 = {
            add_to_list = byzantium_provinces_list
            hidden_effect = { change_province_name = "Byzantium" }
        }
        p:345 = { add_to_list = byzantium_provinces_list }
        p:341 = { add_to_list = byzantium_provinces_list }
        set_variable = change_to_byzantium
        switch_player_country_setup_effect = {
            RULER_NAME = Brasus
            RULER_LAST_NAME = Tsiricus
            RULER_CULTURE = bithynian
            RULER_RELIGION = roman_pantheon
        }
        trigger_event = me_bithynia.5999
    }
}

    type = country_event
    hidden = yes
    immediate = {
        switch_player_country_effect = {
            NEW_NAME = byzantium_name
            NEW_ADJECTIVE = byzantium_adj
            OLD_COUNTRY_COLOR = bithynia_color
            OLD_COUNTRY_NAME = BIT
            OLD_COUNTRY_ADJ = BIT
            OLD_COUNTRY_FLAG = BIT
            OLD_COUNTRY_HERITAGE = bithynia_heritage
            OLD_COUNTRY_CULTURE = bithynian
            OLD_COUNTRY_RELIGION = roman_pantheon
        }
        switch_player_country_second_effect = {
