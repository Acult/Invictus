me_belgae_ultimatum.2 = {		#
    type = country_event
    title = me_belgae_ultimatum_2_title
    desc = me_belgae_ultimatum_2_desc
    picture = revolt
    
    left_portrait = current_ruler

    option = {      
        name = me_belgae_ultimatum_1.a
        every_in_list = {
            list = belgae_subject_list
            if = {
                limit = { 
                    has_variable = accepted_belgae_ultimatum 
                    NOT = { is_subject_of = root }
                }
                root = {
                    make_subject = {
                        type = vassal_tribe
                        target = prev
                    }
                }
            }
        }
        if = {
            limit = {
                any_in_list = {
                    list = belgae_subject_list
                    NOT = { has_variable = accepted_belgae_ultimatum }
                    NOT = { is_subject_of = root }
                }
            }
            custom_tooltip = me_epirus_2_magna.17.at
            hidden_effect = {
                every_in_list = {
                    list = belgae_subject_list
                    limit = {
                        NOT = { is_subject_of = root }
                    }
                    every_owned_province = { add_claim = root }
                }
            }
        }
    }
}
